// Name of the cache const CACHE_NAME = "mineit-cache-v1"; // Files to cache for offline use const ASSETS = [ "./", "./index.html", "./manifest.json", "./style.css", // if CSS moved to external file "./script.js", // if JS moved to external file "https://cdn.jsdelivr.net/npm/chart.js", "https://fonts.googleapis.com/css2?family=Poppins:wght@400;600;700&display=swap", "https://upload.wikimedia.org/wikipedia/commons/thumb/4/46/Bitcoin.svg/1200px-Bitcoin.svg.png", "https://cdn-icons-png.flaticon.com/512/634/634736.png", "https://www.soundjay.com/buttons/sounds/button-16.mp3" ]; // Install Event → Cache app shell self.addEventListener("install", event => { event.waitUntil( caches.open(CACHE_NAME) .then(cache => { console.log("Caching app shell..."); return cache.addAll(ASSETS); }) ); self.skipWaiting(); }); // Activate Event → Clean old caches self.addEventListener("activate", event => { event.waitUntil( caches.keys().then(keys => { return Promise.all( keys.map(key => { if (key !== CACHE_NAME) { console.log("Deleting old cache:", key); return caches.delete(key); } }) ); }) ); self.clients.claim(); }); // Fetch Event → Serve from cache, fallback to network self.addEventListener("fetch", event => { event.respondWith( caches.match(event.request).then(response => { // Return cache if available, else fetch return response || fetch(event.request).catch(() => { // Offline fallback for navigation requests if (event.request.mode === "navigate") { return caches.match("./index.html"); } }); }) ); });
